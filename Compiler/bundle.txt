int nCr(int n, int r){

    if(n < r){
        return 0;
    }
    elif(n == r){
        return 1;
    }
    else{
        if(r==0){
            return 1;
        }
        return nCr(n-1, r-1) + nCr(n-1, r); 
    }
    return 0;
}

int main(){

    output("Enter n: ");
    int n;
    input(n);

    output("Enter r: ");
    int r;
    input(r);

    int res = nCr(n, r);

    output("nCr: ");
    output(res);
    output("\n");
    return 0;
}
